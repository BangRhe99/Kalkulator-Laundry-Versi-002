/** * ========================================================================= * FILE: Script_Shared_UI.html * FUNGSI: Toast, status simpan, dan handler error umum. * BATAS: Tidak berisi logic HPP, Fixed Cost, Harga Layanan, atau rumus fitur. * ========================================================================= */ function showToast(message, kind) { var el = document.getElementById("toast"); if (!el) return; el.textContent = message || ""; el.className = "toast show" + (kind ? " " + kind : ""); window.clearTimeout(showToast._t); showToast._t = window.setTimeout(function () { el.className = "toast"; }, 3200); } // Alias nama pendek untuk pemanggilan UI umum bila dibutuhkan modul baru. function toast(message, kind) { showToast(message, kind); } function setSaveStatus(text, kind) { var el = document.getElementById("saveStatus"); if (!el) return; el.textContent = text || ""; el.className = "save-status" + (kind ? " " + kind : ""); } function clearSaveStatus() { setSaveStatus("", ""); } /** * Satu pintu untuk menampilkan error backend ke user. * Stage tetap dicatat di console supaya debugging lebih cepat. */ function handleBackendError(res, actionLabel) { var msg = (res && res.error) ? res.error : "Terjadi kesalahan tidak dikenal."; var stage = (res && res.stage) ? res.stage : "unknown_stage"; var label = actionLabel || "memproses data"; console.error("[Kalkulator Laundry] Gagal " + label + " (stage: " + stage + "): " + msg); showToast("Gagal " + label + ": " + msg, "err"); setSaveStatus("Gagal", "err"); }
